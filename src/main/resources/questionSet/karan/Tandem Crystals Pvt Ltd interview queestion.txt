public class Main
{
	public static void main(String[] args) {
	    
	    List<Integer> listData = Arrays.asList(1,-1,0,2,-3);
	    List<List<Integer>> finalList = new ArrayList<>();
	    for(int i = 0; i< listData.size(); i++){
	       for(int j = i; j<listData.size(); j++){
	            for(int k = j; k< listData.size();k++){
	             
	             List<Integer> data = new ArrayList<>();
	             if((listData.get(i) + listData.get(j) + listData.get(k)) == 0){
	                 
	             data.add(listData.get(i));
	             data.add(listData.get(j));
	             data.add(listData.get(k));
	                 
	             }   
	               if(data.size()>0){
	                finalList.add(data);
	               }
	            }   
	       }
	  
	    }
	    
		System.out.println(finalList);
	}
}
