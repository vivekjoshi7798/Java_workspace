get the department id, name , salary having highest salary in each department


Select e.DepartmentId, e.Name As EmployeeName, e.salary
from Employee e
join( Select DeptId, Max(salary) As maxsalary
from Employee
group By DepartmentId) as max_salary
On e.departmentId = max_salary.departmentId
& e.salary = max_salary.maxSalary;
