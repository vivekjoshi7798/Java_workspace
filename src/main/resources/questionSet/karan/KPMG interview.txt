public void checkaverageAndprint(List<Integer> marks){

	int length = marks.size();
	int sum = 0;
	for(Integer value: marks){
		sum = sum + value;
	}
	
	int avg = sum/length;
	
	if(avg>60){
	system.out.println("Grage:"+"Pass");
	}
	else{
	system.out.println("Grage:"+"Fail");
	}

}
