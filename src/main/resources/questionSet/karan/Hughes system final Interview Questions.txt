public static int[] getCombination(int[] array, int target){

	Set<Intger> found = new HashSet<>();
	int [] output = new int[2];
	
	for(int num : array){
	
	int subtraction = target - number;
	if(found.contains(subtraction){
	
		output[0] = num;
		output[1] = subtraction;
		
		return output;
	
	}
	
	found.add(num);
	}
	
	output[0] = -1;
	return output;

}
